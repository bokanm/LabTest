#include <LiquidCrystal.h>
#include <util/delay.h>

LiquidCrystal lcd(A0, A1, A2, A3, A4, A5);

byte karo [8] = {
B00000 ,
B00100 ,
B01110 ,
B11111 ,
B01110 ,
B00100 ,
B00000 ,
};

byte tref [8] = {
B00000 ,
B01110 ,
B01110 ,
B11111 ,
B11111 ,
B00100 ,
B00100 ,
};

byte pik [8] = {
B00000 ,
B00100 ,
B01110 ,
B11111 ,
B11111 ,
B00100 ,
B00100 ,
};

byte srce [8] = {
B00000 ,
B01010 ,
B11111 ,
B11111 ,
B01110 ,
B00100 ,
B00000 ,
};


char ocitaj_taster()
{
  char taster = 0;
  unsigned char row;
  
  MCUCR &= ~0x10;
  PORTD = 0x0f; 
  
  for(row = 0x80; row != 0x08; row >>= 1)
  {
    DDRD = row;
	_delay_ms(2);
    
    switch(row | ((~PIND) & 0x0f))
    {
      case 0x88:
      	taster = '1'; break;
      case 0x84:
      	taster = '2'; break;
      case 0x82:
      	taster = '3'; break;
      case 0x81:
      	taster = 'A'; break;
		
	  case 0x48:
      	taster = '4'; break;
      case 0x44:
      	taster = '5'; break;
      case 0x42:
      	taster = '6'; break;
      case 0x41:
      	taster = 'B'; break;
		
	  case 0x28:
      	taster = '7'; break;
      case 0x24:
      	taster = '8'; break;
      case 0x22:
      	taster = '9'; break;
      case 0x21:
      	taster = 'C'; break;
		
	  case 0x18:
      	taster = '*'; break;
      case 0x14:
      	taster = '0'; break;
      case 0x12:
      	taster = '#'; break;
      case 0x11:
      	taster = 'D'; break;
    }
  }
  
  return taster;
}

int main()
{
  	lcd.begin(16, 2);
  	lcd.createChar(0, tref);
  	lcd.createChar(1, srce);
  	lcd.createChar(2, pik);
  	lcd.createChar(3, karo);
  
    unsigned char tacna_kombinacija [4];
  	unsigned char tmp = 0;
	unsigned char kolona = 0;
  	unsigned char taster = 0;
	
	while(1)
    { 
      lcd.home();
	  lcd.print("Pritisni *");
      unsigned long seed = 0;
      lcd.setCursor(0,1);
      while (ocitaj_taster() != '*')
        seed++;
      randomSeed(seed);
      lcd.clear();
      
      for (char i = 0; i < 4; i++) {
      	lcd.setCursor(i,0);
        lcd.write(byte((random(0,4))));
      }
      _delay_ms(2000);
      lcd.clear();
      lcd.cursor();
      
      while (ocitaj_taster() != '5') {
        taster = ocitaj_taster();
        if (taster == '6') kolona++;
        if (kolona == 4) kolona=0;
        if (taster == '4') kolona--;
        if (kolona==255) kolona=3;
        lcd.setCursor(kolona,0);
        
    }
  
  	return 0;
    } }
